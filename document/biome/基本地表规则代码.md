### 基本要素

1. 基岩层

```java
public class MaterialRule {
    public MaterialRules.MaterialRule rule() {
        MaterialRules.MaterialRule BEDROCK = MaterialRules.block(Blocks.BEDROCK.getDefaultState());
        MaterialRules.MaterialCondition transition = MaterialRules.verticalGradient("minecraft:bedrock_floor", YOffset.fixed(0), YOffset.fixed(5));
        return MaterialRules.condition(bedrockSurface, BEDROCK);
    }
}
```
2. 深板岩层

```java
public class MaterialRule {
```

3. 地表层
```java
public class MaterialRule {
    public MaterialRules.MaterialRule rule() {
        MaterialRules.MaterialRule GRASS_BLOCK = MaterialRules.block(Blocks.GRASS_BLOCK.getDefaultState());
        MaterialRules.MaterialRule DIRT = MaterialRules.block(Blocks.DIRT.getDefaultState());
        MaterialRules.MaterialRule grassSurface = MaterialRules.condition(MaterialRules.stoneDepth(0, false, 0, VerticalSurfaceType.FLOOR), GRASS_BLOCK);
        MaterialRules.MaterialRule dirtSurface = MaterialRules.condition(MaterialRules.stoneDepth(4, false, 0, VerticalSurfaceType.FLOOR), GRASS_BLOCK);
        return MaterialRules.sequence(GRASS_BLOCK, DIRT);
    }
}
```
